Tufted.js is data visualization for the modern web. Designed for urban planners, policy professionals, and data geeks with some coding experience, this tool makes it easy to create mobile-friendly, style-agnostic interactive charts and maps. Additionally, because it's extensible and reusable (using Bocoup's d3.chart: http://bocoup.com/weblog/reusability-with-d3/), it makes it easier for developers to optimize the library for particular projects while creating a bridge across an organization's web products. Ultimately, we hope Tufted.js fills a/the missing link between the data analyst and the developer.

Dependencies: D3.js D3chart.js

Development: Download the repo and edit index.html. For live reload, use beefy: beefy build.js:tufted.js --live

Optional: beefy

Build: browserify build.js > tufted.js

Data: Tufted uses CartoDB's SQL API which allows users to quickly visualize data. Start an account at CartoDB.com. 



Charts:
Bar Chart:


  var url = "http://regionalindicators.cartodb.com/api/v2/sql?q=SELECT%20community_type%20AS%20name,%20employment%20AS%20value%20FROM%20emp_by_year_comm_type%20LIMIT%205";

    d3.json(url, function(d) {
      var chart = d3.select('#bar-chart')
        .append("svg")
        .chart("BarChart")
        .yFormat('.2s')
        .height(400);
      //testing some negative values
      d.rows[0].value = d.rows[0].value*-1;

      chart.draw(d.rows);
    });




Grouped Bar Chart:


    var url = "http://regionalindicators.cartodb.com/api/v2/sql?q=SELECT%20max(msa_name)%20AS%20series,%20msa_id,%20json_agg(row_to_json((SELECT%20r%20FROM%20(SELECT%20year AS name,labor_force_of_25_64_year_olds_1%20AS%20value%20)%20r%20WHERE%20labor_force_of_25_64_year_olds_1%20IS%20NOT%20NULL))ORDER%20BY%20year%20ASC)%20AS%20values%20FROM%20labor_force_participation_msa%20WHERE%20labor_force_of_25_64_year_olds_1%20IS%20NOT%20NULL%20GROUP%20BY%20msa_id LIMIT 4";

    d3.json(url, function(d) {
      var chart = d3.select('#grouped-bar-chart')
        .append('svg')
        .chart('GroupedBarChart')
        .yAxisLabel("Percent")
        .height(400)
        .colors(['#F37B70', '#1C164E', '#1F77B4', '#00AFB2']);
      chart.draw(d.rows);    
    });



  
Line Chart:

  
    var url = "http://regionalindicators.cartodb.com/api/v2/sql?q=SELECT%20max(msa_name)%20AS%20series,%20msa_id,%20json_agg(row_to_json((SELECT%20r%20FROM%20(SELECT%20year,labor_force_of_25_64_year_olds_1%20AS%20value%20)%20r%20WHERE%20labor_force_of_25_64_year_olds_1%20IS%20NOT%20NULL))ORDER%20BY%20year%20ASC)%20AS%20values%20FROM%20labor_force_participation_msa%20WHERE%20labor_force_of_25_64_year_olds_1%20IS%20NOT%20NULL%20GROUP%20BY%20msa_id"

    d3.json(url, function(d) {
      var chart = d3.select('#line-chart')
        .append('svg')
        .chart('LineChart')
        .colors(d3.scale.category10())
        .callouts(["Chicago MSA","Boston MSA"])
        .height(400);

      chart.draw(d.rows);  
    });

Public selections: To call D3 functions on existing selections, use layer(""). For example, BarChart contains a "bars" layer. To access that selection, simply use chart.layer("bars").

Other options: All charts come with some optional helper methods. These include: 1. .colors([""]) 2. .height() 3. .width() 4. .yAxis()
